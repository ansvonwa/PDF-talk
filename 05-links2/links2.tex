\documentclass[a4paper,10pt]{article}
\usepackage[utf8]{inputenc}






\immediate\pdfobj{<<
  /S/JavaScript/JS (
  app.alert('Document open!');
  click=function(arg){app.alert('Clicked button ' + arg);})
>>}
\edef\click{\the\pdflastobj\space 0 R}

\pdfnames{
  /JavaScript <<
    %name tree mapping doc. level JavaScript functions to object numbers
    /Names [(click) \click]
    /Limits [(click) (click)] %first and last in alphab. order
  >>
}
